	"fmt"
	"runtime"
	"time"
)

	name string
}

	fmt.Println("Cleaning up resource:", r.name)
}

func main() {
	res := &Resource{"Database Connection"}

	// 设置 res 的终结器
	runtime.SetFinalizer(res, func(r *Resource) {
		r.cleanup()
